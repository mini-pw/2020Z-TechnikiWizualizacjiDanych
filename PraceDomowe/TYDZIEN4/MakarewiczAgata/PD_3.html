Wykres do poprawy

Wykres przedstawia zmiany wspczynnika aktywnoci zawodowej ludnoci wybranych pastw wiata w latach 2000-2018, z podziaem wzgldem pci.

Na wykresie, linia podpisana OECD pokazuje redni warto wskanika pastw nalecych do OECD, czyli Organizacji Wsppracy Gospodarczej i Rozwoju, skupiajcej 36 wysoko rozwinitych, demokratycznych krajw.

Wykres nie zaczyna si od 0, chocia przy tak duych wartociach procentowych jest to zrozumiae. Zaburzona jest skala osi y - jedna z nich obejmuje wartosci 73-88 %, gdzie kolejne linie oznaczaj wartoci rnice sie o 3 %, podczas gdy druga z nich wartosci 55-80 %, ze skokiem co 5 %. Pokazanie tych dwch wykresw obok siebie, z takimi skalami, oraz z liniami na tym samym poziomie, lecz oznaczajcymi rne wartoci, moe atwo wprowadzic obserwatora w bd. Wykres sugeruje, e wskaniki akywnoci zawodowej mczyzn i kobiet s w miare na tym samym poziomie, lub tez minimalnie rni si, podczas gdy tak naprawd rznica jest znaczna, w niektrych pastwach rzdu 15-20 %, co bdzie widoczne na poprawionym wykresie.

Dane do wykresu zostay pobrane ze strony OECD: https://stats.oecd.org/index.aspx?queryid=64197#.

Przygotowanie danych

# wczytanie danych
data <- read.csv2('data.csv',sep=",")

# wybranie danych potrzebnych do wykresw
data <- data%>%
  select(Country,Sex,Time,Value)
data <- transform(data, Value = as.numeric(Value))

# mczyni
data_men <- data%>%
  filter(Sex=="Men")%>%
  ungroup()%>%
  select(-Sex)

# kobiety
data_women <- data%>%
  filter(Sex=="Women")%>%
  ungroup()%>%
  select(-Sex)

# lista wszystkich pastw 
countries <- t(unique(data["Country"]))

# lata
years <- unique(data["Time"])

# przeksztacamy dane, tak aby kada kolumna odpowiadaa innemu pastwu, wiersze to lata

# mezczyzni
a <- 1
b <- dim(years)[1]
data_final_men <- years
while (b<=dim(data_men)[1]){
  data_temp <- slice(data_men["Value"],a:b)
  data_final_men <- cbind(data_final_men,data_temp) 
  a <- a+19
  b <- b+19
}
colnames(data_final_men)[2:4] <- countries[1:3]
colnames(data_final_men)[5:8] <- c("New.Zealand","UK","US","OECD")

# kobiety
a <- 1
b <- dim(years)[1]
data_final_women <- years
while (b<=dim(data_women)[1]){
  data_temp <- slice(data_women["Value"],a:b)
  data_final_women <- cbind(data_final_women,data_temp) 
  a <- a+19
  b <- b+19
}
colnames(data_final_women)[2:4] <- countries[1:3]
colnames(data_final_women)[5:8] <- c("New.Zealand","UK","US","OECD")

Poprawiony wykres

# ustalamy podziak i etykiety na osiach
y_breaks <- seq(55,90, by=5)
y_labels <- c(y_breaks[1:7],"%")
x_breaks <- seq(2000,2018,by=3)

# wykres dla mczyzn
plot1 <- ggplot(data_final_men, aes(x=Time)) + 
  
  # dodajemy lini dla kadego z pastw, przypisujemy nazw do ktrej pniej przypisany zostanie kolor
  geom_line(aes(y = Australia, col="Australia"),  size=1) + 
  geom_line(aes(y = Germany,  col="Germany"),size=1)+
  geom_line(aes(y = Japan,  col="Japan"), size=1)+
  geom_line(aes(y = New.Zealand, col="New Zealand"), size=1)+
  geom_line(aes(y = UK, col="United Kingdom"), size=1)+
  geom_line(aes(y = US, col="United States"), size=1)+
  geom_line(aes(y = OECD, col="OECD"), size=1)+
  
  # ustalamy podziak i skal osi
  scale_x_continuous(breaks = x_breaks)+
  scale_y_continuous(limits=c(55,90),
                     breaks = y_breaks,
                     labels = y_labels)+
  
  # ustalamy etykiety osi i tytu
  ylab("")+
  xlab("Year")+
  ggtitle("Males")+
  
  # dodajemy legend
  scale_color_manual(name="Countries",
                     values=c("Australia"="red",
                              "Germany"="yellow",
                              "Japan"="darkorange",
                              "New Zealand"="darkgrey",
                              "United Kingdom"="purple",
                              "United States"="green",
                              "OECD"="blue"))+
  
  # biae to 
  theme_bw()+
  
  # usuwamy kreseczki na osi y
  theme(axis.ticks.y=element_blank())
  


# analogicznie jak w wykresie pierwszym, pomijamy tworzenie legendy bo jest ona taka sama jak na pierwszym wykresie
