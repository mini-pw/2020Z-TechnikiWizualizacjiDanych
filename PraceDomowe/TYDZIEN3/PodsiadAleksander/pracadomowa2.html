Modyfikacja danych:

# ulatwienie dla wyznaczenia pozycji tekstow
data <- ronaldo %>% mutate(position = case_when(type == 'games' ~ games - 2.3,
                                                type == 'goals' ~ goals + games - 2.3) ) 

# dla poprawnej kolejnosci zmiennych w legendzie
data$type <- factor(data$type, levels = c("games", "goals"))

Modyfikacja wykresu:

Modyfikujc oryginalny wykres miaem na celu przede wszystkim popraw czytelnoci danych. Dziki zastosowaniu geom_bar mona teraz zobaczy nie tylko liczb goli, czy gier Cristiano Ronaldo, ale te ich sum pokazujc ogln produktywno pikarza w danym sezonie. Geom_bar ma rwnie tak zalet, e jest atwo odczytywalny dla ludzkiego oka, ktre najlepiej radzi sobie z porwnywaniem dugoci na wykresach. Poza tym zmodyfikowaem palet kolorw na przyjemniejsz (mniej kontrastow) dla ludzkiego oka. Odwrcenie wykresu rozwizao problem sabej czytelnoci sezonu i roku (tekst nie nachodzi ju na siebie i na wykres). Zastosowanie geom_text pozwala na dokadn analiz postpw i statystyk pikarza (na powyszym wykresie odczytanie tych danych mogo sprawia problemy).

ggplot(data, aes(fill= type, y=value, x=date)) + 
  # odwrocony geom_bar
  geom_bar(position=position_stack(reverse = TRUE), stat="identity") +
  
  # zmiana palety kolorow
  scale_fill_manual(values = c("#303A52", "#596174")) +
  
  theme_classic() +
  
  # dodanie tekstu z liczbami dla lepszej czytelnosci
  geom_text(aes(x = date, y = position, label = value), color = "white") +
  
  # zmiana pozycji legendy
  theme(legend.position = "top") +
  
  # odwrocenie wykresu
  coord_flip() +
  
  # pozbywam sie niepotrzebnych danych na osiach
  theme(axis.line = element_blank(), axis.ticks = element_blank(), axis.text.x = element_blank(),
      legend.title = element_blank(), axis.title = element_blank(), plot.title = element_text(face = "bold"),
      axis.text.y = element_text(face = "bold"), legend.text = element_text(face = "bold")) +
  
  # dodanie tytulu
  labs(title = "Ronaldo games and goals per season")
